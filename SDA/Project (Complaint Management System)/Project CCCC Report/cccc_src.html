           }

           }
           }

           }
           }

           userRole = dir;
           break;
           }
           }
           }
          
           // User role found,   perform the task
           if (  userRole ) {
           userRole->performTask(  depts,   teachers );
           }
           else {
           cout << "Invalid username or password. Please try again.\n";
           }
